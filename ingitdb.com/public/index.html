Quick Start

# Build the CLI
go build -o ingitdb ./cmd/ingitdb

# Validate a database directory
ingitdb validate

# List all collections
ingitdb list collections

# Search records
ingitdb find --substr=Dublin

# Create a new record
ingitdb create record --id=geo/nations/ie --data='{title: "Ireland"}'

# Read a record
ingitdb read record --id=geo/nations/ie
